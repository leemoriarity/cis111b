12   import java.lang.Math;
13   import java.util.*;
14   import java.lang.Comparable;
15   /**
16    * Note Class
17    */
18   public class Note implements Comparable<Note> { // implement Comparable interface
19       private Map<Integer,String> piano = new LinkedHashMap<>(); // declare Map to hold LinkedHashMap values
20       private int key, midi; // key value & midi note value
21       private String note, length, keyType; // musical note (ex: A1), length of note (ex: whole note), & key type (ex: sharp)
22       private double frequency, bpm, noteLength; // frequency of note $ beats per minute
23   
24       /**
25        * Default Note Constructor with no parameters
26        */
27       public Note() {
28           setNote(0);  // set default note to A2
29           setMidiNote(69); // A2 Midi note
30           setKey(0); // key value 0 (A2)
31           setBPM(60); // set default bpm to 60
32           setLength(60,1000); // set length to quarter note
33           setFreq(0); // set frequency of A2
34           setKeyType("A2"); // set key type of A2
35       }
36   
37       /**
38        * @param key key value of pressed key
39        * @param bpm value of beats per minute set by user
40        * @param ms duration of key press in milliseconds
41        */
42       public Note(int key, double bpm, long ms ) {
43           setNote(key);
44           setMidiNote(key+69);
45           setKey(key-69);
46           setBPM(bpm);
47           setLength(bpm,ms);
48           setFreq(key-69);
49           setKeyType(note);
50       }
51   
52       /**
53        * Set Methods
54        * @param key key value of pressed key
55        */
56       public void setNote(int key) {
57           // add key numbers and values sequence to LinkedHashMap
58           piano.put(1,"C");
59           piano.put(2,"C#Db");
60           piano.put(3,"D");
61           piano.put(4,"D#Eb");
62           piano.put(5,"E");
63           piano.put(6,"F");
64           piano.put(7,"F#Gb");
65           piano.put(8,"G");
66           piano.put(9,"G#Ab");
67           piano.put(10,"A");
68           piano.put(11,"A#Bb");
69           piano.put(12,"B");
70           int n = key; // get original key value
71           if (key > 12) // if key is greater than 12
72               key = key%12 + 1; // find key by using key mod 12 + 1
73           note = piano.get(key) + (n/12 - 3); // get key and octave
74       }
75   
76       public void setMidiNote(int key) { midi = key - 69; }
77       public void setKey(int key) { this.key = key; }
78       public void setBPM(double bpm) { this.bpm = bpm; }
79       public void setLength(double bpm, long ms) {
80           long bpm2ms = 60000 / (long)bpm; // get quarter note value in milliseconds for specified bpm
81           if (ms <= bpm2ms / 64) { // if note is less than or equal to Five hundred twelfth note
82               length = "Five hundred twelfth note (" + ms + "ms) goal: " + bpm2ms / 128 + "ms"; // define length and milliseconds including official note length
83               noteLength = 1;
84           } else if (ms >= bpm2ms / 64 && ms < bpm2ms / 32) { // if length is greater than or equal to 1/256 note amd less than 1/128 note
85               length = "Two hundred fifty-sixth note (" + ms + "ms) goal: " + bpm2ms / 64 + "ms";// define length and milliseconds including official note length
86               noteLength = 2;
87           } else if (ms >= bpm2ms / 32 && ms < bpm2ms / 16) { // if length is greater than or equal to 1/128 note amd less than 1/64 note
88               length = "Hundred twenty-eighth note (" + ms + "ms) goal: " + bpm2ms / 32 + "ms";// define length and milliseconds including official note length
89               noteLength = 3;
90           } else if (ms >= bpm2ms / 16 && ms < bpm2ms / 8) { // if length is greater than or equal to 1/64 note amd less than 1/32 note
91               length = "Sixty-fourth note (" + ms + "ms) goal: " + bpm2ms / 16 + "ms";// define length and milliseconds including official note length
92               noteLength = 4;
93           } else if (ms >= bpm2ms / 8 && ms < bpm2ms / 4) { // if length is greater than or equal to 1/32 note amd less than 1/16 note
94               length = "Thirty-second note (" + ms + "ms) goal: " + bpm2ms / 8 + "ms";// define length and milliseconds including official note length
95               noteLength = 5;
96           } else if (ms >= bpm2ms / 4 && ms < bpm2ms / 2) { // if length is greater than or equal to 1/16 note amd less than 1/8 note
97               length = "Sixteenth note (" + ms + "ms) goal: " + bpm2ms / 4 + "ms";// define length and milliseconds including official note length
98               noteLength = 6;
99           } else if (ms >= bpm2ms /2 && ms < bpm2ms) { // if length is greater than or equal to 1/8 note amd less than 1/4 note
100              length = "Eighth note (" + ms + "ms) goal: " + bpm2ms / 2 + "ms";// define length and milliseconds including official note length
101              noteLength = 7;
102          } else if (ms >= bpm2ms && ms < bpm2ms * 2) { // if length is greater than or equal to 1/4 note amd less than 1/2 note
103              length = "Quarter note (" + ms + "ms) goal: " + bpm2ms + "ms";// define length and milliseconds including official note length
104              noteLength = 8;
105          } else if (ms >= bpm2ms * 2 && ms < bpm2ms * 4) { // if length is greater than or equal to 1/2 note amd less than whole note
106              length = "Half note (" + ms + "ms) goal: " + bpm2ms * 2 + "ms";// define length and milliseconds including official note length
107              noteLength = 9;
108          } else { // if length is greater than or equal to whole note
109              length = "Whole note (" + ms + "ms) goal: " + bpm2ms * 4 + "ms";// define length and milliseconds including official note length
110              noteLength = 10;
111          }
112      }
113      public void setFreq(int n) { frequency = 440 * Math.pow(2,n/12F); } // calculate frequency
114      public void setKeyType(String note) {
115          if (note.contains("#")) //if key is black
116              keyType = "Sharp"; // key type is sharp
117          else // if key is white
118              keyType = "Normal"; // key type is normal
119      }

121      /**
122       * Get Methods
123       */
124      public String getNote() {  return note; } // @return note of key
125      public int getMidiNote() {  return midi; } // @return midi note
126      public int getKey() { return key; } // @return key value
127      public int getBPM() { return (int)bpm; } // @return bpm as int
128      public String getLength() { return length; } // @return length of note
129      public double getNoteLength() { return noteLength; }
130      public double getFreq() { return frequency; } // @return frequency formatted to 2 decimal places
131      public String getKeyType() { return keyType; } // @return key type
132  
133      /**
134       * toString Method
135       * @return all information about the key pressed by user
136       */
137      public String toString() {
138  
139          return "Note: " + getNote() +
140                  "\nMidi note: " + getMidiNote() +
141                  "\nKey value: " + getKey() +
142                  "\nKey number: " + (getKey()+22) +
143                  "\nBPM: " + getBPM() +
144                  "\nLength: " + getLength() +
145                  "\nFrequency: " + getFreq() +
146                  "\nKey type: " + getKeyType();
147      }

149      /**
150       * @param n
151       * @return compare result
152       */
153      @Override
154      public int compareTo(Note n) {
155          if (this.frequency > n.getFreq()) // if this note is greater than compared note
156              return 1; // return 1
157          else if (this.frequency < n.getFreq()) // if this note is less than compared note
158              return -1; // return -1
159          else return 0; // if note is equal return 0
160      }

162      /**
163       * @param notes
164       * @return sorted Note List
165       */
166      public List<Note> sortNotes(List<Note> notes){
167          Collections.sort(notes, Comparator.comparing(Note::getNoteLength).thenComparing(Note::getFreq)); // sort list using Collections class and Comparator interface
168          return notes;
169      }
